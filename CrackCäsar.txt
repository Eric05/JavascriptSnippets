function moveChar( offset, char) {

   if ( char == " ") { return " ";} 
   if ( char == "*") { return "*";} 
    var n = char.charCodeAt(0) - 'a'.charCodeAt();
    n = (n + offset) % 26; 
    
    return String.fromCharCode(n + 'a'.charCodeAt());

}

function encString(str,offset) {

var arr = str.split('');
var erg = arr.map(moveChar.bind(null,offset));
var encStr = erg.join('');
return encStr;
}
